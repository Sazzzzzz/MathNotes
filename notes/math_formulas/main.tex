\begin{alignat*}{3}
    \arcsinh'x & =\frac{1}{\sqrt{x^{2}+1}} & \quad &
    \longleftrightarrow \quad & \arcsin'x &
    =\frac{1}{\sqrt{1-x^{2}}}  \\
    \arccosh'x & =\frac{1}{\sqrt{x^{2}-1}} &       &
    \longleftrightarrow       & \arccos'x &
    =-\frac{1}{\sqrt{1-x^{2}}} \\
    \arctanh'x & =\frac{1}{1-x^{2}}        &       &
    \longleftrightarrow       & \arctan'x & =\frac{1}{x^{2}+1}
\end{alignat*}
